**以下记录的是我的创新工场一面经历**

一面时主要问的是算法，问了我两个算法，如下所示：

>一、输入字符串：AB123abd32HDU, 输出：12233ABDHUabd  时间复杂度：O（N）

>我写的代码如下：

	#include <iostream>
	using namespace std;
	int main(){
		char str[100];
		int hash[63]={0};
		cout<<"input:";
		cin>>str;
		for (int i=0;i<strlen(str);i++)
		{
			if ((str[i]<'0'||str[i]>'9')&&(str[i]<'A'||str[i]>'Z')&&(str[i]<'a'||str[i]>'z'))
			{
				cout<<"invalid input!\n";
				break;
			}
			if (str[i]>='0'&&str[i]<='9')
			{
				hash[str[i]-'0']++;
			}
			if (str[i]>='A'&&str[i]<='Z')
			{
				hash[str[i]-'A'+10]++;
			}
			if (str[i]>='a'&&str[i]<='z')
			{
				hash[str[i]-'a'+36]++;
			}
		}
		int j=0;
		cout<<"\noutput:";
		for (i=0;i<10;i++)
		{
			while (hash[i]>0)
			{
				str[j++]=i+'0';
				hash[i]--;
			}
		}
		
		for (i=10;i<36;i++)
		{
			while (hash[i]>0)
			{
				str[j++]=i-10+'A';
				hash[i]--;
			}
		}
		for (i=36;i<62;i++)
		{
			while (hash[i]>0)
			{
				str[j++]=i-36+'a';
				hash[i]--;
			}
		}
		cout<<str<<endl;
		return 0;
	}
